```c#
public class Obstacle : MonoBehaviour {
float delta = -0.1f;

//Start 메소드 생략

void Update() {
 float newXPosition = transform.position.x + delta;
 transform.position = new Vector3(newXPosition, 2, -7); //핵심

 if( transform.position.x < -3.5 )
 {
  delta = 0.1f;
 }
 else if( transform.position.x > 3.5 )
 {
  delta = -0.1f;
 }
} 
}
```
